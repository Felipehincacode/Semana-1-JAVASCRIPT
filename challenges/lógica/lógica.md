// Se inicializan a null para indicar que a√∫n no tienen un valor.
let lado1 = null;

/**
 * Funci√≥n para solicitar al usuario las longitudes de los tres lados del tri√°ngulo.
 * Utiliza la funci√≥n 'prompt' para obtener la entrada y la convierte a tipo Number.
 */
function pedirdatos() {
  lado1 = Number(prompt("Ingrese la longitud del primer lado "));
  lado2 = Number(prompt("Ingrese la longitud del segundo lado "));
  lado3 = Number(prompt("Ingrese la longitud del tercer lado "));
}

/**
 * Funci√≥n para verificar si los lados proporcionados pueden formar un tri√°ngulo v√°lido.
 * @param {number} a - Longitud del primer lado.
 * @param {number} b - Longitud del segundo lado.
 * @param {number} c - Longitud del tercer lado.
 */
function verificar(a, b, c) {
  // Comprueba si todas las longitudes de los lados son n√∫meros positivos.
  if (a > 0 && b > 0 && c > 0) {
    // Si los lados son positivos, verifica la regla del tri√°ngulo:
    // La suma de dos lados debe ser mayor que el tercer lado.
    if ((a + b) > c && (b + c) > a && (c + a) > b) {
      // Si los lados forman un tri√°ngulo v√°lido, lo indica y procede a clasificarlo.
      console.log(`los valores ${a} ${b} ${c} son correctos`);
      clasificar(a, b, c); // Se pasan los par√°metros a la funci√≥n clasificar
    } else {
      // Si los lados no forman un tri√°ngulo v√°lido (pero son positivos), informa el error
      // y solicita nuevamente los datos.
      console.info("hay un error en los datos ingresado, por favor intentelo de nuevo");
      pedirdatos();
      // Se debe volver a llamar a verificar con los nuevos lados
      verificar(lado1, lado2, lado3);
    }
  } else {
    // Si alguna longitud de lado no es positiva, informa el error
    // y solicita nuevamente los datos.
    console.info("hay un error en los datos ingresado, por favor intentelo de nuevo");
    pedirdatos();
    // Se debe volver a llamar a verificar con los nuevos lados
    verificar(lado1, lado2, lado3);
  }
}

/**
 * Funci√≥n para clasificar el tipo de tri√°ngulo (Equil√°tero, Is√≥sceles, Escaleno).
 * @param {number} a - Longitud del primer lado.
 * @param {number} b - Longitud del segundo lado.
 * @param {number} c - Longitud del tercer lado.
 */
function clasificar(a, b, c) {
  // Comprueba si todos los lados son iguales (Equil√°tero).
  if ((a == b) && (b == c)) {
    console.log("El triangulo es Equilatero");
  }
  // Comprueba si al menos dos lados son iguales (Is√≥sceles).
  // Esta condici√≥n se eval√∫a si no es equil√°tero.
  else if (a == b || b == c || c == a) {
    console.log("El Triangulo es Is√≥sceles ");
  }
  // Si no es equil√°tero ni is√≥sceles, todos los lados son diferentes (Escaleno).
  else {
    console.log("El Triangulo es Escaleno");
  }
}
